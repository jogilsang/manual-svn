### SVN 서버 윈도우세팅
svnadmin create C:\svn\window

sc create svn binpath= "\"C:\Program Files (x86)\Subversion\bin\svnserve.exe\" --service -r C:\svn\window" displayname= "Subversion Server" depend= Tcpip start= auto
sc delete svn
